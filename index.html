export default { async fetch(request, env) { // Chỉ cho phép POST if (request.method !== "POST") { return new Response("Only POST allowed", { status: 405 }); } // Lấy nội dung gửi từ website const body = await request.json(); const message = body.message; // Gọi Gemini API const res = await fetch( "https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=" + env.GEMINI_API_KEY, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [ { parts: [ { text: "Bạn là chatbot tư vấn bảo hiểm xã hội Việt Nam. Trả lời ngắn gọn, đúng luật, không bịa.\nCâu hỏi: " + message } ] } ] }) } ); const data = await res.json(); // Lấy câu trả lời const reply = data?.candidates?.[0]?.content?.parts?.[0]?.text || "Xin lỗi, tôi chưa có câu trả lời phù hợp."; // Trả dữ liệu về website return new Response( JSON.stringify({ reply }), { headers: { "Content-Type": "application/json" } } ); } };
